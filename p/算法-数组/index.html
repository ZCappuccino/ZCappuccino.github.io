class Solution{
    public int removeDuplicates(int[] nums){
        int p=0,q=1;
        for(q=1;q < nums.length;q++){
            if(nums[p]!=nums[q]){
		nums[p+1]=nums[q];
		p++;
	    }
	}
	return p+1;
    }
}
